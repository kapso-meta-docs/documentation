```
import requests

url = "https://graph.facebook.com/&lt;API_VERSION&gt;/&lt;WHATSAPP_BUSINESS_PHONE_NUMBER_ID&gt;/messages"
headers = {
    "Authorization": "Bearer &lt;ACCESS_TOKEN&gt;",
    "Content-Type": "application/json",
}
data = {
    "messaging_product": "whatsapp",
    "to": "&lt;WHATSAPP_USER_PHONE_NUMBER&gt;",
    "type": "template",
    "template": {
        "name": "hello_world",
        "language": {"code": "en_US"},
    }
}

response = requests.post(url, headers=headers, json=data, timeout=30)
print(response.json())

```
